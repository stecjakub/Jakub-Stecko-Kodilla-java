SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS POSTS_NUMBER
FROM USERS U JOIN POSTS P ON U.ID = P.USER_ID
GROUP BY P.USER_ID
HAVING COUNT(*) > 1
ORDER BY U.LASTNAME, U.FIRSTNAME;

SELECT * FROM ISSUES;

SELECT I2.NAME, I.SUMMARY
FROM ISSUESLISTS I2
JOIN ISSUES I ON I2.ID = I.ISSUESLIST_ID;

SELECT U.FIRSTNAME, U.LASTNAME, I.SUMMARY
FROM USERS U
JOIN ISSUES I on U.ID = I.USER_ID_ASSIGNEDTO;

SELECT USERS.FIRSTNAME, USERS.LASTNAME, COUNT(*) AS ISSUES_NUMBER
FROM USERS
JOIN ISSUES I on USERS.ID = I.USER_ID_ASSIGNEDTO
GROUP BY I.USER_ID_ASSIGNEDTO;
